# 控制流
```
x <- runif(5)
x
# for循环
for (i in x) {
  print(i)
}
# while循环
i=1
while (i <= 5) {
  print(x[i]) ; i <- i+1
}
# 条件(矢量化的)
ifelse(x>.5,"大","小")
switch("2",
  "1" = "hello",
  "2" = "hi"
  )
```
